### K个一组翻转链表  
给你一个链表，每 k 个节点一组进行翻转，请你返回翻转后的链表。  
k 是一个正整数，它的值小于或等于链表的长度。  
如果节点总数不是 k 的整数倍，那么请将最后剩余的节点保持原有顺序。  
**思路：**
1.建立一个判断是否可以翻转的函数。  
2.建立一个翻转k个节点的函数，返回翻转后的头节点，队列q保存翻转后的尾节点。  
3.主函数将每个翻转后的链表连接到一起。  
```c++
queue<ListNode*> q; //保存每一次翻转后的尾节点，用于指向下一个翻转后的头节点
ListNode* temp; //指向下一个未翻转的头节点
ListNode* reverseKGroup(ListNode* head, int k) {
    	if(!canReverse(head,k))z
        return head;
        ListNode* cur = head;
        //保留第一段翻转后的表头
        ListNode* res = reverse(cur,k);
        cur = temp;
        while(canReverse(cur,k))
        {
            //q头部是上一段翻转后的尾结点，指向下一段翻转后的头结点
            ListNode* pre = q.front();
            pre->next = reverse(cur,k);
            q.pop();
            cur = temp;
        }
        //下一段不能翻转，上一段的尾结点直接指向下一段的头结点
        ListNode* pre = q.front();
        pre->next = cur;
        return res;

    }

    //选取k个结点翻转,返回翻转后的头节点，q保留反转后的尾结点
    ListNode* reverse(ListNode* l, int k)
    {
        ListNode* cur = l;
        q.push(cur);
        ListNode* Next = cur->next;
        int m = k - 1;
        while(cur && Next && m)
        {
            temp = Next->next;
            Next->next = cur;
            cur = Next;
            Next = temp;
            m = m - 1;
        }
        return cur;
    }
    
    //能否翻转
    bool canReverse(ListNode* l,int k)
    {
        if(k == 1)
            return false;
        ListNode* cur = l;
        int n = k;
        while(cur&&n)
        {
            cur = cur->next;
            n--;
        }
        if(n>0)
            return false;
        else 
            return true;
    } 
```
