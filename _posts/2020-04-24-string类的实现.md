利用char* 实现string的构造函数，复制函数，析构函数，赋值函数，拼接函数，长度函数以及输出函数。  
注：用strcpy()编译时出错，换成strcpy_s(char * des, length, const char* src)，length = strlen(src) + 1。  

```c++
#ifndef STRING_H
#define STRING_H
#include<cstring>
#include<string.h>
#include<iostream>
using namespace std;
class String
{
public:
	//默认构造函数
	String();
	//构造函数
	String(const char* str);
	//复制构造函数
	String(String &str);
	//析构函数
	~String();
	//字符串拼接
	String operator+(const String& str);
	//字符串赋值
	String& operator=(const char* str);
	bool operator==(const String& str);
	//字符串长度
	int length();
	//重载输出
	friend ostream& operator<<(ostream& o, const String& str);
private:
	char* m_data;
	int m_size;
};
#endif // !STRING_H

```

```c++
#include"String.h"

//构造函数
String::String(const char* str)
{
	if (str == NULL)
	{
		m_data = new char[1];
		m_data[0] = '/0';
		m_size = 1;
	}
	else
	{
		m_size = strlen(str);
		m_data = new char[m_size + 1];
		strcpy_s(m_data, m_size + 1, str);
	}
}
//复制构造函数
String::String(String& str)
{
	m_size = str.m_size;
	m_data = new char[m_size + 1];
	strcpy_s(m_data, m_size + 1, str.m_data);
}
//析构函数
String::~String()
{
	delete[] m_data;
	m_size = 0;
}
//拼接字符串
String String::operator+(const String& str)
{
	//新建一个String对象用于返回
	String newStr;
	//释放其空间
	delete[] newStr.m_data;
	newStr.m_size = m_size + str.m_size;
	newStr.m_data = new char[newStr.m_size + 1];
	//拼接到一块
	strcpy_s(newStr.m_data, m_size + 1, m_data);
	strcpy_s(newStr.m_data + m_size, str.m_size + 1 , str.m_data);
	return newStr;
}
//字符串赋值
String& String::operator=(const char* str)
{
	if (m_data == str)
		return *this;
	delete[] m_data;
	m_size = strlen(str);
	m_data = new char[m_size + 1];
	strcpy_s(m_data, m_size + 1, str);
	return *this;
}
bool String::operator==(const String& str)
{
	return strcmp(m_data, str.m_data) == 0;
}
//字符串长度
int String::length()
{
	return m_size;
}
//输出重载
ostream& operator<<(ostream& o, const String& str)
{
	o << str.m_data;
	return o;
}
```
